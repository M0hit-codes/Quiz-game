"""
Project 4: The Quiz Game

Goal:
    Ask the user 3 questions, check their answers, track a score,
    and show a final result.

Key Skills:
    If/Elif/Else conditions & Score Tracking.

How the question bank is structured:
    A list of dictionaries - connecting Project 1 (lists) and the
    dictionary concept introduced there. Each dictionary is one row:
        "question" : the question text
        "options"  : list of choices shown to the user
        "answer"   : the correct option number as a string ("1", "2"...)

    Storing questions this way means adding a new question is just
    adding one more dictionary to the list - the rest of the code
    handles it automatically.
"""

# ── Question Bank ────────────────────────────────────────────────
# List of dicts - the same structure a database table uses, just
# held in memory instead of on disk.

QUESTIONS = [
    {
        "question": "What is the capital of France?",
        "options": ["Berlin", "Paris", "Madrid", "Rome"],
        "answer": "2"
    },
    {
        "question": "Which data type stores multiple items in Python?",
        "options": ["Integer", "Boolean", "List", "Float"],
        "answer": "3"
    },
    {
        "question": "What keyword is used to define a function in Python?",
        "options": ["function", "define", "fun", "def"],
        "answer": "4"
    },
    {
        "question": "What does int('50') do?",
        "options": [
            "Raises an error",
            "Returns the string '50'",
            "Converts '50' to the integer 50",
            "Returns 0"
        ],
        "answer": "3"
    },
    {
        "question": "Which module provides cryptographically secure random choices?",
        "options": ["random", "math", "secrets", "string"],
        "answer": "3"
    },
]

VALID_OPTIONS = {"1", "2", "3", "4"}


# ── Display helpers ──────────────────────────────────────────────

def show_question(number, total, question_data):
    """
    OUTPUT: print one question with its numbered options.
    Keeping display separate from evaluation - the same
    Model/View split the training material talks about.
    """
    print(f"\nQuestion {number} of {total}:")
    print(question_data["question"])
    for index, option_text in enumerate(question_data["options"], start=1):
        print(f"  {index}. {option_text}")


def get_answer(question_data):
    """
    INPUT + validation: keep asking until the user picks a valid
    option number. Accepts "1", "2", "3", or "4" only.

    Using numbered options instead of free-text avoids the
    case-sensitivity problem entirely - "2" is always "2".
    """
    valid_count = str(len(question_data["options"]))
    valid_set = {str(n) for n in range(1, len(question_data["options"]) + 1)}

    while True:
        user_input = input(f"Your answer (1-{valid_count}): ").strip()

        if user_input in valid_set:
            return user_input

        print(f"Please enter a number between 1 and {valid_count}.\n")


def evaluate_answer(user_answer, question_data):
    """
    PROCESS (the evaluation engine): compare the user's choice to
    the stored correct answer using if/elif/else.

    This is the core new skill of Project 4 - conditional logic
    that makes a decision and returns a result.
    """
    if user_answer == question_data["answer"]:
        correct_option = question_data["options"][int(user_answer) - 1]
        print(f"Correct! The answer is: {correct_option}")
        return True
    else:
        correct_index = int(question_data["answer"]) - 1
        correct_option = question_data["options"][correct_index]
        print(f"Wrong. The correct answer was: {correct_option}")
        return False


def show_final_result(score, total):
    """
    OUTPUT: display the final score and a performance message.
    The score display format matches what the brief specifies:
    "You got X out of Y correct."
    """
    print("\n" + "=" * 36)
    print(f"  You got {score} out of {total} correct.")

    # if/elif/else chain - tiered feedback based on score
    if score == total:
        print("  Perfect score! Outstanding work.")
    elif score >= total * 0.6:
        print("  Good effort - keep practising!")
    else:
        print("  Keep going - review and try again!")

    print("=" * 36)


# ── Main ─────────────────────────────────────────────────────────

def main():
    total = len(QUESTIONS)

    print("\n" + "-" * 36)
    print("           QUIZ GAME")
    print("-" * 36)
    print(f"Answer all {total} questions. Enter the number of your choice.\n")

    # Score tracker initialized OUTSIDE the loop - same accumulator
    # pattern as Project 2. Inside the loop would reset it every question.
    score = 0

    for question_number, question_data in enumerate(QUESTIONS, start=1):
        show_question(question_number, total, question_data)
        user_answer = get_answer(question_data)

        # evaluate_answer returns True/False; adding True/False to an
        # integer works in Python because bool is a subclass of int
        # (True == 1, False == 0) - cleaner than an extra if block.
        score += evaluate_answer(user_answer, question_data)

    show_final_result(score, total)


if __name__ == "__main__":
    main()
